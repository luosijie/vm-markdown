### Install

```bash
npm install --save vm-markdown
```
### Usage

```js
import VmMarkdown from 'vm-markdown'
export default {
  ...
  components: {
  	VmMarkdown
  },
  methods: {
  	showHtml (html) {
      // get html string here
    },
    showRaw (raw) {
      // get raw string here
    }
  }
  ...
}
```

```vue
<VmMarkdown theme="default" //dark, green, gray, princess 
            width="1000px" 
            height="600px" 
            v-on:gethtml="showHtml"
            v-on:getraw="showRaw"
            :defaultText="intro">
</VmMarkdown>
```
